import java.math.BigInteger;
import java.util.Scanner;


public class uva10070_st {// for submit change it to "Main"
	public static void main(String args[]){
		Scanner ab=new Scanner(System.in);
		
		BigInteger b1;
		int i=0;
		
		while(ab.hasNextBigInteger()){
			b1=ab.nextBigInteger();
			if(i!=0)
				System.out.println();
			i=1;
			if((b1.mod(BigInteger.valueOf(400)).equals(BigInteger.ZERO))||((b1.mod(BigInteger.valueOf(4)).equals(BigInteger.ZERO))&&!(b1.mod(BigInteger.valueOf(100)).equals(BigInteger.ZERO)))){
				System.out.println("This is leap year.");
				if(b1.mod(BigInteger.valueOf(15)).equals(BigInteger.ZERO))
					System.out.println("This is huluculu festival year.");
				if(b1.mod(BigInteger.valueOf(55)).equals(BigInteger.ZERO))
					System.out.println("This is bulukulu festival year.");
				
			}
			else if(b1.mod(BigInteger.valueOf(15)).equals(BigInteger.ZERO))
				System.out.println("This is huluculu festival year.");
			else
				System.out.println("This is an ordinary year.");
		}
	}
}
